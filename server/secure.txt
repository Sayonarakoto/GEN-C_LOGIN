<script>
  document.addEventListener('DOMContentLoaded', () => {
    const passkeyForm = document.getElementById('passkeyForm');
    const passkeyInput = document.getElementById('passkey');
    const debugOutput = document.getElementById('debugOutput');

    function updateDebugOutput(message) {
      debugOutput.innerHTML = `<p>${message}</p>`;
    }

    passkeyForm.addEventListener('submit', async (event) => {
      event.preventDefault();
      const passkey = passkeyInput.value;

      // Update URL to new backend endpoint with port 3001
      const url = '/api/security/submit-passkey';

      const data = { passkey };
      updateDebugOutput("Submitting passkey... please wait.");
      console.log("Submitting request to:", url);
      console.log("Request body:", data);

      try {
        const response = await fetch(url, {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json'
          },
          body: JSON.stringify(data)
        });

        if (!response.ok) {
          throw new Error(`HTTP error! status: ${response.status}`);
        }

        const resBody = await response.json();
        console.log("Backend response (res.body):", resBody);
        updateDebugOutput(`Success! Passkey submitted.\nBackend message: ${JSON.stringify(resBody, null, 2)}`);

      } catch (error) {
        console.error("Fetch error:", error);
        updateDebugOutput(`Submission failed! Check the console for more details.\nError: ${error.message}`);
      }
    });
  }); 
</script> </html>
