// Sidebar Link (used in sidebar)
function SidebarLink({ icon, text, active }) {
  const color = active ? "#3b82f6" : "#6b7280";
  const w600 = active ? 600 : 500;
  return (
    <Box
      sx={{
        display: "flex", alignItems: "center",
        px: 3, py: 1.5,
        bgcolor: active ? "#e0f2fe" : "inherit",
        color: color,
        fontWeight: w600,
        borderRadius: 2,
        mb: 1,
        transition: "0.2s"
      }}
    >
      <i className={`bx bx-${icon}`} style={{ fontSize: 24, marginRight: 14, color }} />
      <Typography sx={{ display: { xs: "none", md: "inline" }, color, fontWeight: w600 }}>{text}</Typography>
    </Box>
  );
}

// Dashboard Stat Card
function DashboardStatCard({ title, value, color }) {
  return (
    <Card sx={{ borderRadius: 3, p: 3 }}>
      <Typography fontSize={14} color="#64748b" gutterBottom fontWeight={500}>{title}</Typography>
      <Typography fontSize={32} fontWeight={700} sx={{ mt: 1, color }}>{value}</Typography>
    </Card>
  );
}

// Each student request card in the Approval Queue
function StudentRequestCard({ name, id, date, reason, status, updated, statusColor }) {
  let chipProps = {};
  if (status === "Pending") chipProps = { sx: { bgcolor: "#fef9c3", color: "#f59e42", fontWeight: 700, fontSize: 13 } };
  else if (status === "Approved") chipProps = { sx: { bgcolor: "#dcfce7", color: "#22c55e", fontWeight: 700, fontSize: 13 } };
  else chipProps = { sx: { bgcolor: "#fee2e2", color: "#ef4444", fontWeight: 700, fontSize: 13 } };

  return (
    <Card sx={{ borderRadius: 2, border: "1px solid #e5e7eb", p: 2, mb: 2 }}>
      <Box sx={{ display: "flex", justifyContent: "space-between", alignItems: "center", mb: 1.5 }}>
        <Typography variant="h6" fontWeight={600} color="#111827">{name}</Typography>
        <Chip label={status} {...chipProps} />
      </Box>
      <Typography fontSize={14} color="#4b5563" mb={0.5}><b>ID:</b> {id}</Typography>
      <Typography fontSize={14} color="#4b5563" mb={0.5}><b>Date:</b> {date}</Typography>
      <Typography fontSize={14} color="#4b5563" mb={1}><b>Reason:</b> {reason}</Typography>
      <Typography fontSize={12} color="#9ca3af" mb={2}><i>Last Updated: {updated}</i></Typography>
      <Box sx={{ display: "flex", gap: 2 }}>
        <MUIButton
          sx={{
            flex: 1,
            bgcolor: status === "Pending" ? "#22c55e" : "#e5e7eb",
            color: status === "Pending" ? "#fff" : "#6b7280"
          }}
          disabled={status !== "Pending"}
        >
          Approve
        </MUIButton>
        <MUIButton
          sx={{
            flex: 1,
            bgcolor: status === "Pending" ? "#ef4444" : "#e5e7eb",
            color: status === "Pending" ? "#fff" : "#6b7280"
          }}
          disabled={status !== "Pending"}
        >
          Reject
        </MUIButton>
      </Box>
    </Card>
  );
}

// Bottom mobile nav
function FooterNavItem({ icon, text, active }) {
  return (
    <Box sx={{ display: "flex", flexDirection: "column", alignItems: "center", gap: 0.5, color: active ? "#3b82f6" : "#6b7280" }}>
      <i className={`bx bx-${icon}`} style={{ fontSize: 24, fontVariationSettings: active ? "'FILL' 1" : undefined }} />
      <Typography sx={{ fontSize: 13, fontWeight: 400 }}>{text}</Typography>
    </Box>
  );
}
